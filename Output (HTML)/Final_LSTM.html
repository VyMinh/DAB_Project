def create_dataset(dataset, time_step=1):
    """
    To mt dataset t chui thi gian, s dng ca s trt (sliding window).
    
    Parameters:
        dataset: Mt mng 1 chiu cha cc gi tr chui thi gian
        time_step: S bc thi gian dng  d bo bc tip theo
        
    Returns:
        dataX: Mng cc on chui u vo, mi on c  di time_step
        dataY: Mng cc gi tr d on tng ng vi mi on u vo
    """
    dataX, dataY = [], []
    for i in range(len(dataset) - time_step):
        # Ly on time_step gi tr lin tip lm u vo X
        a = dataset[i:(i + time_step)]
        dataX.append(a)
        # Gi tr tip theo lm u ra Y
        dataY.append(dataset[i + time_step])
    return np.array(dataX), np.array(dataY)

import tensorflow as tf

# nh ngha hm xy dng m hnh vi hyperparameter tuning
def build_model(hp):
    model = Sequential()
    
    # Lp LSTM u tin vi L2 regularization mnh hn
    model.add(LSTM(
        units=hp.Int('units_1', min_value=32, max_value=128, step=32),  # Gim s units ti a
        return_sequences=False,  # Simplify model
        input_shape=(time_step, 1),
        activation=hp.Choice('activation', values=['relu', 'tanh']),
        recurrent_regularizer=tf.keras.regularizers.l2(hp.Float('recurrent_l2', min_value=1e-5, max_value=1e-3, sampling='log')),  # Thm regularization cho recurrent weights
        kernel_regularizer=tf.keras.regularizers.l2(hp.Float('l2', min_value=1e-5, max_value=1e-3, sampling='log'))
    ))
    
    # Dropout cao hn
    model.add(Dropout(hp.Float('dropout_1', min_value=0.3, max_value=0.5, step=0.1)))  # Tng dropout ti thiu
    
    # Lp u ra 
    model.add(Dense(1))
    
    # Compile m hnh vi learning rate c iu chnh
    optimizer_choice = hp.Choice('optimizer', values=['adam', 'rmsprop'])
    learning_rate = hp.Float('learning_rate', min_value=1e-5, max_value=1e-3, sampling='log')
    
    if optimizer_choice == 'adam':
        optimizer = tf.keras.optimizers.Adam(learning_rate=learning_rate)
    else:
        optimizer = tf.keras.optimizers.RMSprop(learning_rate=learning_rate)
    
    model.compile(
        optimizer=optimizer,
        loss='mean_squared_error'
    )
    return model
# Ci t Keras Tuner
