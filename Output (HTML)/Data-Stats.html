2. Kim tra mi quan h phi tuyn tnh (Mutual Information)

Gii thch: Mutual Information (MI) kim tra mc lin h gia Price v cc ct khc, k c khi mi lin h khng tuyn tnh.

In[2]:
from sklearn.feature_selection import mutual_info_regression

# Chn bin u vo v mc tiu
X = data[['Open', 'High', 'Low', 'Vol.']]
y = data['Price']

# Tnh Mutual Information
mi_scores = mutual_info_regression(X, y)

# Hin th kt qu
mi_scores = pd.Series(mi_scores, index=X.columns)
mi_scores.sort_values(ascending=False, inplace=True)

# V biu  Mutual Information
plt.figure(figsize=(8, 5))
mi_scores.plot(kind='bar', color='teal')
plt.title('Mutual Information Scores with Price')
plt.ylabel('Score')
plt.show()

6. Thng k xu hng gi vng theo cc ngy trong tun

In[6]:
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# c li d liu
data = pd.read_csv('./Data/Gold_Futures_Cleaned.csv')  # Thay bng ng dn file ca bn

# Chuyn ct 'Date' sang datetime v to ct 'Day_of_Week'
data['Date'] = pd.to_datetime(data['Date'])
data['Day_of_Week'] = data['Date'].dt.day_name()

# Tnh cc thng k theo ngy trong tun
stats_by_day = data.groupby('Day_of_Week')['Price'].agg(['mean', 'median', 'max', 'min'])

# m bo th t cc ngy trong tun
day_order = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday']
stats_by_day = stats_by_day.reindex(day_order)

# Vit ha nhn cc ngy trong tun
day_labels_vn = ['Th Hai', 'Th Ba', 'Th T', 'Th Nm', 'Th Su']

# V biu  nhm ct
x = np.arange(len(day_order))  # V tr ca cc ngy trong tun
width = 0.2  #  rng ca mi nhm ct

plt.figure(figsize=(12, 6))
plt.bar(x - width * 1.5, stats_by_day['mean'], width, label='Trung bnh', color='skyblue')
plt.bar(x - width * 0.5, stats_by_day['median'], width, label='Trung v', color='orange')
plt.bar(x + width * 0.5, stats_by_day['max'], width, label='Cao nht', color='green')
plt.bar(x + width * 1.5, stats_by_day['min'], width, label='Thp nht', color='red')

# Gn nhn v thm thng tin
plt.title('Thng k xu hng gi vng theo cc ngy trong tun', fontsize=14)
plt.xlabel('Cc ngy trong tun', fontsize=12)
plt.ylabel('Gi vng', fontsize=12)
plt.xticks(x, day_labels_vn)  # S dng nhn ting Vit
plt.legend()
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.tight_layout()
plt.show()

import pandas as pd
import matplotlib.pyplot as plt

# Load li file bng pandas, v d:
gld_price_data = pd.read_csv('./Data/Gold_Futures_Cleaned.csv')

# Kim tra cc ct trong d liu  m bo c ct 'Date'
print(gld_price_data.columns)

# Nu ct 'Date' tn ti, ta s to thm ct 'Year' t ct 'Date'
gld_price_data['Year'] = pd.to_datetime(gld_price_data['Date']).dt.year

# V boxplot so snh gi vng theo nm
plt.figure(figsize=(12, 8))
gld_price_data.boxplot(column='Price', by='Year', grid=False, figsize=(12, 6), showmeans=True)
plt.title('Gi vng phn phi theo nm')
plt.suptitle('')  # Xa tiu  mc nh ca pandas boxplot
plt.xlabel('Nm')
plt.ylabel('Gi vng')
plt.xticks(rotation=45)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()
